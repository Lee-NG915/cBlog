> 这种实现方式需要在应用中提供Provider, 才能在客户端访问, 无法通过`NEXT_PUBLIC_` 实现直接注入客户端.

## 实现

可以在应用启动前加载json配置进ENV中.

```
// env.json

{
  "GTM_ID": "GTM-XXXXXX",
  "API_HOST": "https://https://api-test.castlery.sg",
  "NEXT_PUBLIC_API": "https://api-test.castlery.sg",
  "US": {
    "CURRENCY": "USD",
    "TIME_ZONE": "America/New_York",
    "othes": {
      "CURRENCY": "USD",
      "TIME_ZONE": "America/New_York"
    }
  },
  "SG": {
    "CURRENCY": "SGD",
    "TIME_ZONE": "Asia/Singapore"
  },
  "AU": {
    "CURRENCY": "AUD",
    "TIME_ZONE": "Australia/Sydney"
  }
}
```

```
// nextjs instrumentation

import envJSON from './.env.json';

// 将JSON路径转换为字符串路径
// { US : { currency: "USD"}} => US.currency

function transformEnv(env: Record<string, any>, prefix = '') {
  for (const key in env) {
    const value = env[key];
    if (typeof value === 'object') {
      transformEnv(value, prefix + key + '.');
    } else {
      process.env[prefix + key] = value;
    }
  }
}

export function register() {
  const env = envJSON as Record<string, any>;
  transformEnv(env);
  console.log(process.env);
}
```



使用

```
process.env['US.xxx']
process.env['SG.xxx']
const country = process.env['country']
process.env[`${country}.xxxx`]
```
